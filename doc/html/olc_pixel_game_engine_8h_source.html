9  What is this?
10  ~~~~~~~~~~~~~
11  olc::PixelGameEngine is a single file, cross platform graphics and userinput
12  framework used for games, visualisations, algorithm exploration and learning.
13  It was developed by YouTuber "javidx9" as an assistive tool for many of his
14  videos. The goal of this project is to provide high speed graphics with
15  minimal project setup complexity, to encourage new programmers, younger people,
16  and anyone else that wants to make fun things.
17 
18  However, olc::PixelGameEngine is not a toy! It is a powerful and fast utility
19  capable of delivering high resolution, high speed, high quality applications
20  which behave the same way regardless of the operating system or platform.
21 
22  This file provides the core utility set of the olc::PixelGameEngine, including
23  window creation, keyboard/mouse input, main game thread, timing, pixel drawing
24  routines, image/sprite loading and drawing routines, and a bunch of utility
25  types to make rapid development of games/visualisations possible.

1029  bool ResourcePack::LoadPack(const std::string& sFile, const std::string& sKey)
1030  {
1031  // Open the resource file
1032  baseFile.open(sFile, std::ifstream::binary);
1033  if (!baseFile.is_open()) return false;
1034 
1035  // 1) Read Scrambled index
1036  uint32_t nIndexSize = 0;
1037  baseFile.read((char*)&nIndexSize, sizeof(uint32_t));
1038 
1039  std::vector<char> buffer(nIndexSize);
1040  for (uint32_t j = 0; j < nIndexSize; j++)
1041  buffer[j] = baseFile.get();
1042 
1043  std::vector<char> decoded = scramble(buffer, sKey);
1044  size_t pos = 0;
1045  auto read = [&decoded, &pos](char* dst, size_t size) {
1046  memcpy((void*)dst, (const void*)(decoded.data() + pos), size);
1047  pos += size;
1048  };
1049 
1050  auto get = [&read]() -> int {
1051  char c;
1052  read(&c, 1);
1053  return c;
1054  };
1055 
1056  // 2) Read Map
1057  uint32_t nMapEntries = 0;
1058  read((char*)&nMapEntries, sizeof(uint32_t));
1059  for (uint32_t i = 0; i < nMapEntries; i++)
1060  {
1061  uint32_t nFilePathSize = 0;
1062  read((char*)&nFilePathSize, sizeof(uint32_t));
1063 
1064  std::string sFileName(nFilePathSize, ' ');
1065  for (uint32_t j = 0; j < nFilePathSize; j++)
1066  sFileName[j] = get();
1067 
1068  sResourceFile e;
